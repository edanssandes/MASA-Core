00072         int i0;

00074         int i1;
00075         int j1;

00079                 return true;
00080         }

00088 
00089         //fprintf(stderr, "--isBlockPrunable? (%d,%d) (%d,%d) %d+%d<%d\n", bx, by, i0, j0, score,inc, bestScore);
00090 
00091 
00092         if (recurrenceType == NEEDLEMAN_WUNSCH) {
00093                 int gaps;
00094 
00095                 gaps = abs(distJ-distI) - std::max(j1-j0, i1-i0);
00096                 if (gaps > 0) {
00097                         inc -= score_params->gap_open + gaps*score_params->gap_ext;
00098                 }
00099

00101                 dec -= distMin*score_params->mismatch;

00103                 dec += score_params->gap_open + gaps*score_params->gap_ext;
