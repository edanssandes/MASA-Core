00046 int ReversedCellsReader::read(cell_t* buf, int len) {
00047         if (len > position) {
00048                 len = position;

00050         position -= len;
00051         reader->seek(position);

00054         // Reverse buffer order

00056                 cell_t aux = buf[i];
