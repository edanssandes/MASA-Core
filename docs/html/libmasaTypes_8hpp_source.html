00026 /**
00027  * Struct that represents a cell in the DP matrix. See the Smith-Waterman
00028  * recurrence function with the gotoh affine gap modification.
00029  *
00030  * Each cell has three components: H, E and F. But note that the E and F
00031  * components holds the same memory area, because the E and F components are a
00032  * union. So, the cell structure may represent only two components simultaneously,
00033  * the \f$(H,F)\f$ components or the \f$(H,E)\f$ components.
00034  */
00035 typedef struct {
00036         int h;
00037         union {
00038                 int f;
00039                 int e;
00040         };
00041 } __attribute__ ((aligned (8))) cell_t;
