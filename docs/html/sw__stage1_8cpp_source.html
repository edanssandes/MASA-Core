00068 static score_t getInitialBestScore(int alignment_start, int alignment_end) {
00069         int may_start_at_corner_00 = alignment_start == AT_ANYWHERE
00070                         || alignment_start == AT_SEQUENCE_1
00071                         || alignment_start == AT_SEQUENCE_2
00072                         || alignment_start == AT_SEQUENCE_1_OR_2
00073                         || alignment_start == AT_SEQUENCE_1_AND_2;
00074         int may_start_at_corner_01 = alignment_start == AT_ANYWHERE
00075                         || alignment_start == AT_SEQUENCE_1
00076                         || alignment_start == AT_SEQUENCE_1_OR_2;
00077         int may_start_at_corner_10 = alignment_start == AT_ANYWHERE
00078                         || alignment_start == AT_SEQUENCE_2
00079                         || alignment_start == AT_SEQUENCE_1_OR_2;
00080         int may_start_at_corner_11 = alignment_start == AT_ANYWHERE;
00081 
00082         int may_end_at_corner_00 = alignment_end == AT_ANYWHERE;
00083         int may_end_at_corner_01 = alignment_end == AT_ANYWHERE
00084                         || alignment_end == AT_SEQUENCE_2
00085                         || alignment_end == AT_SEQUENCE_1_OR_2;
00086         int may_end_at_corner_10 = alignment_end == AT_ANYWHERE
00087                         || alignment_end == AT_SEQUENCE_1
00088                         || alignment_end == AT_SEQUENCE_1_OR_2;
00089         int may_end_at_corner_11 = alignment_end == AT_ANYWHERE
00090                         || alignment_end == AT_SEQUENCE_1
00091                         || alignment_end == AT_SEQUENCE_2
00092                         || alignment_end == AT_SEQUENCE_1_OR_2
00093                         || alignment_end == AT_SEQUENCE_1_AND_2;
00094 
00095         score_t best;
00096         best.i = -1;
00097         best.j = -1;
00098         if ((may_start_at_corner_00 && may_end_at_corner_00)
00099                         || (may_start_at_corner_01 && may_end_at_corner_01)
00100                         || (may_start_at_corner_10 && may_end_at_corner_10)
00101                         || (may_start_at_corner_11 && may_end_at_corner_11)) {
00102                 best.score = 0;
00103         } else {
00104                 best.score = -INF;
00105         }
00106         return best;
00107 }
