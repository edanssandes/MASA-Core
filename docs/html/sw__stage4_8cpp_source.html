00298                 return false;

00799         if (new_positions[i].type != -1 && diff_pos) {
00800             has_new_pos = 1;

00806         crosspoints->clear();
00807         crosspoints->assign(merged_partitions.begin(), merged_partitions.end());
00808         //crosspoints->save();
